    <title>希望島 2027：互動設計沙盤 v15</title>
    <script src="https://cdn.jsdelivr.net/npm/react@18.2.0/umd/react.production.min.js"></script>
    <script src="https://cdn.jsdelivr.net/npm/react-dom@18.2.0/umd/react-dom.production.min.js"></script>
    <script src="https://cdn.jsdelivr.net/npm/@babel/standalone@7.23.10/babel.min.js"></script>
    <script src="https://cdn.tailwindcss.com"></script>
    <script src="https://cdn.jsdelivr.net/npm/lucide@0.344.0/dist/umd/lucide.min.js"></script>
    <link href="https://fonts.googleapis.com/css2?family=Noto+Sans+TC:wght@400;700;900&display=swap" rel="stylesheet">
    <style>
        body { background-color: #f1f5f9; font-family: 'Noto Sans TC', sans-serif; height: 100vh; overflow: hidden; margin: 0; }
        .map-bg { 
            background-image: url('Gemini_Generated_Image_fd56hefd56hefd56.png'); 
            background-size: cover; 
            background-position: center; 
            border: 8px solid white;
        }
        .glass { background: rgba(255, 255, 255, 0.85); backdrop-filter: blur(10px); }
        .draggable-item { cursor: grab; transition: all 0.2s; }
        .draggable-item:active { cursor: grabbing; transform: scale(1.1); }
        @keyframes error-blink { 0%, 100% { border-color: transparent; } 50% { border-color: #ef4444; background: rgba(239, 68, 68, 0.2); } }
        .placement-error { animation: error-blink 1s infinite; border: 4px solid #ef4444 !important; }
    </style>

<body>
    <div id="root"></div>
    <script type="text/babel">
        const { useState, useEffect, useRef } = React;

        const PROJECTS = [
            { id: 'off_wind', name: '離岸風機', cost: 1.5, power: 60, storage: 0, icon: 'Wind', color: 'text-teal-500', bg: 'bg-teal-500', zone: 'sea' },
            { id: 'land_wind', name: '陸域風機', cost: 0.5, power: 15, storage: 0, icon: 'Tornado', color: 'text-emerald-600', bg: 'bg-emerald-600', zone: 'land' },
            { id: 'solar_farm', name: '大型光電', cost: 1.2, power: 45, storage: 0, icon: 'Sun', color: 'text-amber-500', bg: 'bg-amber-500', zone: 'land' },
            { id: 'roof_solar', name: '屋頂光電', cost: 0.2, power: 5, storage: 0, icon: 'Home', color: 'text-orange-500', bg: 'bg-orange-500', zone: 'land' },
            { id: 'wave_power', name: '波浪能站', cost: 1.0, power: 25, storage: 0, icon: 'Waves', color: 'text-indigo-600', bg: 'bg-indigo-600', zone: 'sea' },
            { id: 'battery', name: '儲能站', cost: 0.8, power: 0, storage: 40, icon: 'BatteryCharging', color: 'text-rose-500', bg: 'bg-rose-500', zone: 'land' },
            { id: 'smart_grid', name: '智慧電網', cost: 0.4, power: 0, storage: 0, icon: 'Cpu', color: 'text-violet-600', bg: 'bg-violet-600', zone: 'land' }
        ];

        const ROLES = [
            { id: 'eco', name: '生態保育官', emoji: '🦅', target: '禁東岸風機' },
            { id: 'econ', name: '經濟部長', emoji: '💰', target: '產能極大化' },
            { id: 'tour', name: '觀光局長', emoji: '🏖️', target: '西岸美感與靜音' },
            { id: 'village', name: '漁村里長', emoji: '🐟', target: '電力絕對穩定' }
        ];

        const MISSIONS = [
            { id: 'typhoon', name: '強颱襲擊', desc: '風機=0，光電=50%', emoji: '🌪️' },
            { id: 'birds', name: '候鳥遷徙', desc: '夜間產能剩電池/波浪', emoji: '🦢' },
            { id: 'heatwave', name: '觀光熱浪', desc: '日需求提升至 130', emoji: '🔥' },
            { id: 'corrosion', name: '海水鏽蝕', desc: '波浪能產量暫時歸零', emoji: '⚙️' },
            { id: 'chain', name: '零件斷鏈', desc: '離岸風機效率下降 30%', emoji: '🔗' },
            { id: 'eco_order', name: '生態保育令', desc: '東岸設施產量扣除 40%', emoji: '📜' }
        ];

        const Icon = ({ name, size = 20, className = "" }) => {
            const iconRef = useRef(null);
            useEffect(() => {
                if (window.lucide && iconRef.current) {
                    const iconName = name.replace(/([a-z])([A-Z])/g, '$1-$2').toLowerCase();
                    if (lucide.icons[iconName]) iconRef.current.innerHTML = lucide.icons[iconName].toSvg({ strokeWidth: 2, width: size, height: size, class: className });
                }
            }, [name, size, className]);
            return <span ref={iconRef} className="inline-flex"></span>;
        };

        const App = () => {
            const [step, setStep] = useState(1);
            const [role, setRole] = useState(null);
            const [mission, setMission] = useState(null);
            const [budget, setBudget] = useState(5.0);
            const [inventory, setInventory] = useState({});
            const [placedItems, setPlacedItems] = useState([]);
            const [testResult, setTestResult] = useState(null);
            const [quantities, setQuantities] = useState({});

            const resetToHome = () => { setStep(1); setRole(null); setMission(null); setBudget(5.0); setInventory({}); setPlacedItems([]); setTestResult(null); };

            const getEngineStats = () => {
                let totalP = 0; let totalS = 0; let errorCount = 0;
                let hasGrid = placedItems.some(i => i.typeId === 'smart_grid');

                placedItems.forEach(item => {
                    const p = PROJECTS.find(proj => proj.id === item.typeId);
                    const isSea = (item.y < 20 || item.y > 80 || item.x < 20 || item.x > 80);
                    const currentZone = isSea ? 'sea' : 'land';

                    let output = p.power;
                    if (p.zone !== currentZone) { output = 0; errorCount++; }

                    // 地理影響：東岸保育令 (x > 66%)
                    if (mission?.id === 'eco_order' && item.x > 66) { output *= 0.6; }

                    totalP += output;
                    totalS += p.storage;
                });
                if (hasGrid) totalP *= 1.15;
                return { p: Math.round(totalP), s: totalS, errors: errorCount };
            };

            const runPressureTest = () => {
                const stats = getEngineStats();
                const hasGrid = placedItems.some(i => i.typeId === 'smart_grid');
                const gridF = hasGrid ? 1.15 : 1.0;
                let score = 100;
                let feedbacks = []; let pros = []; let cons = [];

                // 1. 基礎需求與預算判定
                const baseDemand = mission.id === 'heatwave' ? 130 : 100;
                if (stats.p < baseDemand && mission.id !== 'heatwave') { 
                    score -= 40; 
                    feedbacks.push(`⚠️ 常態產能不足：平日電力(${stats.p})低於需求的 100 單位。`); 
                    cons.push("基本發電量未達標"); 
                }
                if (budget < 0) { 
                    score = 0; 
                    feedbacks.push("💸 財務破產：預算超支！這是一份不負責任的計畫。"); 
                    cons.push("預算規劃完全失敗"); 
                }

                // 2. 地理錯誤處罰
                if (stats.errors > 0) { 
                    score -= (stats.errors * 10); 
                    feedbacks.push(`❌ 工程錯誤：發現 ${stats.errors} 個設備放置在錯誤地形。`); 
                    cons.push("地形匹配失誤"); 
                }

                // 3. 顧問團角色與設計紅線需求判定 (重構核心)
                // 生態官：東岸 (x > 66) 嚴禁風機
                if (role.id === 'eco') {
                    const offendingItems = placedItems.filter(i => i.x > 66 && i.typeId.includes('wind'));
                    if (offendingItems.length > 0) {
                        score -= 40;
                        feedbacks.push(`🦅 生態官大怒：你在候鳥區架設了 ${offendingItems.length} 座風機，嚴重威脅候鳥安全！`);
                        cons.push("觸碰生態保育紅線");
                    } else {
                        pros.push("完美守護候鳥棲息地");
                    }
                }
                // 觀光局長：西岸 (x < 33) 嚴禁大型機械 (風機、大型光電)
                if (role.id === 'tour') {
                    const offendingItems = placedItems.filter(i => i.x < 33 && (i.typeId.includes('wind') || i.typeId === 'solar_farm'));
                    if (offendingItems.length > 0) {
                        score -= 40;
                        feedbacks.push(`🏖️ 觀光局長不滿：西岸度假區出現了 ${offendingItems.length} 座巨大機械，嚴重破壞海景並產生噪音！`);
                        cons.push("無視觀光景觀價值");
                    } else {
                        pros.push("確保了度假村的頂級視覺環境");
                    }
                }
                // 經濟部長：追求 CP 值，希望電力遠大於需求
                if (role.id === 'econ') {
                    if (stats.p < 120) {
                        score -= 20;
                        feedbacks.push("💰 經濟部長質疑：產電量剛好而已，我們需要更多電力來支撐工業發展！");
                        cons.push("發電餘裕不足");
                    } else {
                        pros.push("強大的電力產能獲經濟部肯定");
                    }
                }
                // 漁村里長：重視儲能穩定性
                if (role.id === 'village') {
                    if (stats.s < 40) {
                        score -= 20;
                        feedbacks.push("🐟 里長擔憂：儲能容量不足，一旦沒風沒太陽，漁獲就完了！");
                        cons.push("系統韌性儲備不足");
                    } else {
                        pros.push("完善的儲能配置深受漁民信賴");
                    }
                }

                // 4. 緊急任務判定 (精準同步簡報算式)
                let survivalP = 0;
                const getPowerByType = (typeIdFilter) => {
                    return placedItems
                        .filter(i => {
                            const p = PROJECTS.find(proj => proj.id === i.typeId);
                            const isSea = (i.y < 20 || i.y > 80 || i.x < 20 || i.x > 80);
                            return i.typeId.includes(typeIdFilter) && p.zone === (isSea ? 'sea' : 'land');
                        })
                        .reduce((acc, i) => acc + PROJECTS.find(proj => proj.id === i.typeId).power, 0) * gridF;
                };

                if (mission.id === 'typhoon') {
                    const windP = getPowerByType('wind');
                    const solarP = getPowerByType('solar');
                    survivalP = Math.round((stats.p - windP - (solarP * 0.5)) + stats.s);
                } 
                else if (mission.id === 'birds') {
                    const waveP = getPowerByType('wave_power');
                    survivalP = Math.round(waveP + stats.s);
                } 
                else if (mission.id === 'heatwave') {
                    survivalP = stats.p;
                }
                else if (mission.id === 'corrosion') {
                    const waveP = getPowerByType('wave_power');
                    survivalP = Math.round((stats.p - waveP) + stats.s);
                }
                else if (mission.id === 'chain') {
                    const offWindP = getPowerByType('off_wind');
                    survivalP = Math.round((stats.p - (offWindP * 0.3)) + stats.s);
                }
                else if (mission.id === 'eco_order') {
                    survivalP = stats.p + stats.s;
                }

                if (survivalP < baseDemand) {
                    score -= 50;
                    feedbacks.push(`🚨 壓力測試失敗：面對「${mission.name}」，全島可用電力僅餘 ${survivalP} 單位，低於需求。`);
                    cons.push("缺乏極端環境適應力");
                } else {
                    pros.push(`成功克服了「${mission.name}」的生存考驗`);
                }

                const finalScore = Math.max(0, score);
                setTestResult({ totalScore: finalScore, feedbacks, pros, cons, survivalP, baseDemand });
            };

            if (step === 1) return (
                <div className="flex flex-col items-center justify-center h-screen bg-slate-900 text-white">
                    <h1 className="text-5xl font-black mb-12 text-teal-400">希望島 2027：任務委任系統</h1>
                    <div className="grid grid-cols-2 gap-8 max-w-5xl">
                        <div className="bg-white/10 p-10 rounded-[50px] border-2 border-white/20 text-center">
                            <h2 className="text-xl font-bold mb-6 text-teal-300">1. 選擇顧問角色</h2>
                            <div className="grid grid-cols-2 gap-3">
                                {ROLES.map(r => <button key={r.id} onClick={()=>setRole(r)} className={`p-4 rounded-2xl border-2 ${role?.id === r.id ? 'bg-teal-500 border-white scale-105' : 'bg-white/5 border-transparent hover:bg-white/10'}`}>{r.emoji} {r.name}</button>)}
                            </div>
                        </div>
                        <div className="bg-white/10 p-10 rounded-[50px] border-2 border-white/20 text-center">
                            <h2 className="text-xl font-bold mb-6 text-rose-300">2. 抽取突發挑戰</h2>
                            <div className="grid grid-cols-2 gap-3">
                                {MISSIONS.map(m => <button key={m.id} onClick={()=>setMission(m)} className={`p-4 rounded-2xl border-2 ${mission?.id === m.id ? 'bg-rose-500 border-white scale-105' : 'bg-white/5 border-transparent hover:bg-white/10'}`}>{m.emoji} {m.name}</button>)}
                            </div>
                        </div>
                    </div>
                    {role && mission && <button onClick={()=>setStep(2)} className="mt-12 bg-teal-400 text-slate-900 px-16 py-4 rounded-full font-black text-2xl shadow-xl active:scale-95 transition">進入沙盤設計</button>}
                </div>
            );

            return (
                <div className="flex h-screen overflow-hidden bg-slate-50">
                    <div className="flex-1 relative flex flex-col p-6 gap-6">
                        <div className="flex justify-between items-center glass p-5 rounded-[32px] shadow-xl border border-white/50">
                            <div className="flex gap-4">
                                <div className="bg-white px-4 py-2 rounded-xl shadow-sm border border-slate-100 font-bold">{role.emoji} {role.name}</div>
                                <div className="bg-rose-50 px-4 py-2 rounded-xl shadow-sm border border-rose-100 text-rose-600 font-bold">{mission.emoji} {mission.name}</div>
                            </div>
                            <div className="flex items-center gap-6">
                                <div className={`text-3xl font-mono font-black ${budget < 0 ? 'text-rose-500' : 'text-teal-600'}`}>{budget} <span className="text-sm">億</span></div>
                                <button onClick={resetToHome} className="bg-slate-800 text-white p-3 rounded-2xl hover:bg-slate-700 transition"><Icon name="Home" size={24}/></button>
                            </div>
                        </div>

                        <div className="flex-1 map-bg rounded-[50px] shadow-2xl relative overflow-hidden" onDragOver={e=>e.preventDefault()} onDrop={(e)=>{
                            const typeId = e.dataTransfer.getData("typeId");
                            if (inventory[typeId] > 0) {
                                const rect = e.currentTarget.getBoundingClientRect();
                                const x = ((e.clientX - rect.left) / rect.width) * 100;
                                const y = ((e.clientY - rect.top) / rect.height) * 100;
                                setPlacedItems([...placedItems, { id: Date.now(), x, y, typeId }]);
                                setInventory({...inventory, [typeId]: inventory[typeId]-1});
                            }
                        }}>
                            {/* 地形引導文字 */}
                            <div className="absolute top-0 left-0 w-full h-[20%] bg-blue-500/10 border-b-2 border-dashed border-blue-400 flex items-center justify-center pointer-events-none"><span className="text-blue-400 font-black text-4xl opacity-40 uppercase tracking-widest">波浪能潛力區 (SEA)</span></div>
                            <div className="absolute bottom-0 left-0 w-full h-[20%] bg-blue-500/10 border-t-2 border-dashed border-blue-400 flex items-center justify-center pointer-events-none"><span className="text-blue-400 font-black text-4xl opacity-40 uppercase tracking-widest">南部海域 (SEA)</span></div>
                            
                            {/* 區域引導：西岸 vs 東岸 */}
                            <div className="absolute top-[20%] left-0 w-[33%] h-[60%] border-r-2 border-dashed border-blue-200 pointer-events-none flex items-center justify-center">
                                <span className="text-blue-900/10 font-black text-6xl -rotate-90">西岸：觀光度假區</span>
                            </div>
                            <div className="absolute top-[20%] right-0 w-[33%] h-[60%] border-l-2 border-dashed border-rose-200 pointer-events-none flex items-center justify-center">
                                <span className="text-rose-900/10 font-black text-6xl rotate-90">東岸：候鳥棲息地</span>
                            </div>

                            {placedItems.map(item => {
                                const p = PROJECTS.find(proj => proj.id === item.typeId);
                                const isSea = (item.y < 20 || item.y > 80 || item.x < 20 || item.x > 80);
                                const hasError = p.zone !== (isSea ? 'sea' : 'land');

                                return <div key={item.id} style={{ left: `${item.x}%`, top: `${item.y}%` }} 
                                    className={`absolute -translate-x-1/2 -translate-y-1/2 p-2 bg-white rounded-2xl shadow-xl border-4 ${hasError ? 'placement-error' : p.color.replace('text-', 'border-')} group transition-all z-10 hover:scale-150`}>
                                    <Icon name={p.icon} className={p.color} size={32} />
                                    <button onClick={()=> { setPlacedItems(placedItems.filter(i=>i.id!==item.id)); setInventory({...inventory, [item.typeId]:(inventory[item.typeId]||0)+1}) }} className="absolute -top-3 -right-3 bg-red-500 text-white rounded-full w-7 h-7 font-black opacity-0 group-hover:opacity-100 flex items-center justify-center shadow-lg transition-opacity">×</button>
                                </div>
                            })}
                        </div>
                    </div>

                    <div className="w-[450px] glass p-10 overflow-y-auto flex flex-col gap-8 shadow-2xl border-l border-slate-200 text-slate-800">
                        <div className="flex justify-between items-center"><h3 className="text-2xl font-black">1. 採購清單</h3><button onClick={()=>{setInventory({});setPlacedItems([]);setBudget(5.0)}} className="text-xs bg-rose-100 text-rose-600 px-4 py-2 rounded-xl font-bold">清空重設</button></div>
                        <div className="space-y-4">
                            {PROJECTS.map(p => (
                                <div key={p.id} className="bg-white p-4 rounded-[32px] border border-slate-100 shadow-sm">
                                    <div className="flex justify-between items-center mb-2">
                                        <div className="flex items-center gap-2"><div className={`w-3 h-3 rounded-full ${p.bg}`}></div><b className="text-sm">{p.name} ({p.cost}億)</b></div>
                                        <div className="text-[10px] bg-slate-50 px-2 rounded-full font-black text-slate-400">庫存: {inventory[p.id]||0}</div>
                                    </div>
                                    <div className="flex gap-2">
                                        <input type="number" placeholder="數量" value={quantities[p.id]||""} onChange={e=>setQuantities({...quantities,[p.id]:e.target.value})} className="w-16 bg-slate-50 rounded-xl px-2 font-bold text-center focus:outline-none" />
                                        <button onClick={()=>{
                                            const q = parseInt(quantities[p.id] || 1);
                                            if (budget >= p.cost * q) { setBudget(prev => Math.round((prev - p.cost * q)*10)/10); setInventory({...inventory, [p.id]: (inventory[p.id]||0) + q}); setQuantities({...quantities, [p.id]: ""}); }
                                        }} className={`${p.bg} text-white font-black flex-1 py-1 rounded-xl text-xs shadow-md hover:brightness-110`}>確認購買</button>
                                    </div>
                                </div>
                            ))}
                        </div>
                        
                        <div className="bg-slate-100 p-6 rounded-[40px] border-4 border-dashed border-slate-200 flex flex-wrap gap-3 items-center justify-center min-h-[140px]">
                            {PROJECTS.map(p => (inventory[p.id]||0) > 0 && <div key={p.id} draggable onDragStart={e=>e.dataTransfer.setData("typeId", p.id)} className={`draggable-item bg-white p-4 rounded-[28px] shadow-xl border-4 ${p.color.replace('text-', 'border-')} relative`}>
                                <Icon name={p.icon} className={p.color} size={28}/><span className="absolute -top-3 -right-3 bg-slate-900 text-white text-[10px] w-7 h-7 rounded-full flex items-center justify-center font-black">x{inventory[p.id]}</span>
                            </div>)}
                            {Object.values(inventory).every(v=>v===0) && <p className="text-sm text-slate-400 font-black italic">採購設備後拖曳至地圖...</p>}
                        </div>

                        <div className="mt-auto pt-8 border-t-2 border-slate-100">
                            <div className="grid grid-cols-3 gap-2 text-center mb-6">
                                <div className="bg-white p-3 rounded-2xl shadow-sm"><div className="text-[10px] opacity-40 font-black">發電量</div><div className="text-xl font-black">{getEngineStats().p}</div></div>
                                <div className="bg-white p-3 rounded-2xl shadow-sm"><div className="text-[10px] opacity-40 font-black">挑戰需求</div><div className="text-xl font-black text-teal-600">{mission.id==='heatwave'?130:100}</div></div>
                                <div className="bg-white p-3 rounded-2xl shadow-sm"><div className="text-[10px] opacity-40 font-black">總儲能</div><div className="text-xl font-black text-rose-500">{getEngineStats().s}</div></div>
                            </div>
                            <button onClick={runPressureTest} className="w-full bg-slate-900 text-white py-5 rounded-[40px] font-black text-xl hover:bg-teal-500 shadow-xl transition-all flex items-center justify-center gap-3">
                                <Icon name="Play" size={24}/>提交設計並審查
                            </button>
                        </div>
                    </div>

                    {testResult && (
                        <div className="fixed inset-0 z-[1000] flex items-center justify-center bg-slate-900/95 backdrop-blur-xl p-10">
                            <div className="bg-white rounded-[60px] p-20 max-w-4xl w-full text-center relative overflow-hidden">
                                <div className={`absolute top-0 left-0 w-full h-8 ${testResult.totalScore >= 60 ? 'bg-teal-500' : 'bg-rose-500'}`}></div>
                                <h2 className="text-5xl font-black mb-8 text-slate-800">顧問團結案報告</h2>
                                <div className="text-9xl font-black text-slate-900 mb-8">{testResult.totalScore} <span className="text-2xl text-slate-300">PTS</span></div>
                                
                                <div className="bg-slate-50 p-6 rounded-3xl border mb-10 text-lg font-bold text-slate-600">
                                    任務狀態：面對 {mission.name}，系統維持了 {testResult.survivalP} 單位電力（需求：{testResult.baseDemand}）
                                </div>

                                <div className="grid grid-cols-2 gap-10 text-left mb-10">
                                    <div className="bg-emerald-50 p-8 rounded-[40px]"><b>優點亮點：</b>{testResult.pros.length > 0 ? testResult.pros.map((p,i)=><div key={i} className="text-sm">✓ {p}</div>) : <div className="text-sm text-slate-400">尚無顯著優點</div>}</div>
                                    <div className="bg-rose-50 p-8 rounded-[40px]"><b>待優化處：</b>{testResult.cons.length > 0 ? testResult.cons.map((c,i)=><div key={i} className="text-sm">✗ {c}</div>) : <div className="text-sm text-emerald-600">設計已相當完備</div>}</div>
                                </div>
                                
                                <div className="bg-slate-100 p-6 rounded-3xl mb-10 italic text-slate-500 font-medium">「 {testResult.feedbacks.join(' ')} 」</div>

                                <div className="flex gap-4">
                                    <button onClick={()=>setTestResult(null)} className="flex-1 bg-slate-100 py-4 rounded-3xl font-black hover:bg-slate-200">返回修改設計</button>
                                    <button onClick={resetToHome} className="flex-1 bg-slate-900 text-white py-4 rounded-3xl font-black shadow-xl">重抽角色任務</button>
                                </div>
                            </div>
                        </div>
                    )}
                </div>
            );
